## Infinite limits

<Observe>
  {({ setState, ...state }) => {
    switch (state.mode) {
      
      case 'regular': return <>
        <BlockMath math="\lim_{x\to 0^+}\frac{1}{x^n}=+\infty" />
        <BlockMath math="\lim_{x\to 0^-}\frac{1}{x^n}=\begin{cases}+\infty&&\text{if $n$ is even}\\ -\infty&&\text{if $n$ is odd}\end{cases}
        " />
        
        If <InlineMath math="\lim_{x\to a}f=\pm\infty" /> and <InlineMath math="\lim_{x\to a}g=\pm c" />:

        <BlockMath math="\lim_{x\to a}(f\pm g)=\begin{cases}+\infty+c=+\infty \\ +\infty-c=+\infty \\ -\infty+c=-\infty \\ -\infty-c=-\infty\end{cases}" />
        <BlockMath math="\lim_{x\to a}(fg)=\begin{cases}(+\infty)(+c)=+\infty \\ (+\infty)(-c)=-\infty \\ (-\infty)(+c)=-\infty \\ (-\infty)(-c)=+\infty\end{cases}" />

        If <InlineMath math="\lim_{x\to a}f=0" /> and <InlineMath math="\lim_{x\to a}g=\pm c" />:

        <BlockMath math="\lim_{x\to a}\frac{g}{f}=\begin{cases}\frac{+c}{0^+}=+\infty \\ \frac{+c}{0^-}=-\infty \\ \frac{-c}{0^+}=-\infty \\ \frac{-c}{0^-}=+\infty\end{cases}" />

      </>;

      case 'color': return <>
        
        <BlockMath math="\lim_{x\to 0^+}\frac{1}{x^{\color{fuchsia}{n}}}=+\infty" />
        <BlockMath math="\lim_{x\to 0^-}\frac{1}{x^{\color{fuchsia}{n}}}=\begin{cases}+\infty&&\text{if ${\color{fuchsia}{n}}$ is even}\\ -\infty&&\text{if ${\color{fuchsia}{n}}$ is odd}\end{cases}
        " />
        
        If <InlineMath math="\lim_{x\to a}{\color{orange}{f}}=\pm\infty" /> and <InlineMath math="\lim_{x\to a}{\color{orange}{g}}=\pm {\color{fuchsia}{c}}" />:

        <BlockMath math="\lim_{x\to a}({\color{orange}{f}}\pm {\color{orange}{g}})=\begin{cases}+\infty+{\color{fuchsia}{c}}=+\infty \\ +\infty-{\color{fuchsia}{c}}=+\infty \\ -\infty+{\color{fuchsia}{c}}=-\infty \\ -\infty-{\color{fuchsia}{c}}=-\infty\end{cases}" />
        <BlockMath math="\lim_{x\to a}({\color{orange}{fg}})=\begin{cases}(+\infty)(+{\color{fuchsia}{c}})=+\infty \\ (+\infty)(-{\color{fuchsia}{c}})=-\infty \\ (-\infty)(+{\color{fuchsia}{c}})=-\infty \\ (-\infty)(-{\color{fuchsia}{c}})=+\infty\end{cases}" />

        If <InlineMath math="\lim_{x\to a}{\color{orange}{f}}=0" /> and <InlineMath math="\lim_{x\to a}{\color{orange}{g}}=\pm {\color{fuchsia}{c}}" />:

        <BlockMath math="\lim_{x\to a}\frac{\color{orange}{g}}{\color{orange}{f}}=\begin{cases}\frac{+{\color{fuchsia}{c}}}{0^+}=+\infty \\ \frac{+{\color{fuchsia}{c}}}{0^-}=-\infty \\ \frac{-{\color{fuchsia}{c}}}{0^+}=-\infty \\ \frac{-{\color{fuchsia}{c}}}{0^-}=+\infty\end{cases}" />
      </>

    }
  }}
</Observe>

## Squeeze theorem / Trigonometry


<div className="grid grid-flow-row md:grid-rows-2 md:grid-flow-col gap-y-3 md:gap-y-3">

$$
\lim_{x\to 0}\frac{\sin x}{x}=1
$$
$$
\lim_{x\to 0}\frac{1-\cos x}{x}=0
$$

$$
\lim_{x\to \infty}\frac{\sin x}{x}=0
$$
$$
\lim_{x\to \infty}\frac{\cos x}{x}=0
