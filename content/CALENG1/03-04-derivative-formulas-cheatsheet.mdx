## Rules

<div className="grid grid-flow-row md:grid-rows-4 md:grid-flow-col gap-y-3 md:gap-0 text-center">

<Observe>
  {({ setState, ...state }) => {
    switch (state.mode) {

      case 'regular': return <>
        <div>Constant rule<br/>
          <BlockMath math="\frac{d}{dx}(n)=0" />
        </div>
        <div>Power rule<br/>
          <BlockMath math="\frac{d}{dx}(x^n)=nx^{n-1}" />
        </div>
        <div>Constant multiple rule<br/>
          <BlockMath math="\frac{d}{dx}(nf)=nf'" />
        </div>
        <div>Chain rule<br/>
          <BlockMath math="\frac{d}{dx}(f(g))=f'(g)g'" />
        </div>

        <div>Sum & difference rule<br/>
          <BlockMath math="\frac{d}{dx}(f \pm g)=f' \pm g'" />
        </div>
        <div>Product rule<br/>
          <BlockMath math="\frac{d}{dx}(fg)=f'g+fg'" />
        </div>
        <div>Quotient rule<br/>
          <BlockMath math="\frac{d}{dx}\left( \frac{f}{g} \right)=\frac{f'g-fg'}{g^2}" />
        </div>
        <div>General power rule<br/>
          <BlockMath math="\frac{d}{dx}\left(f^n\right)=nf^{n-1}f'" />
        </div>
        

      </>;
      case 'color': return <>
        <div>Constant rule<br/>
          <BlockMath math="\frac{d}{dx}({\color{fuchsia}{n}})=0" />
        </div>
        <div>Power rule<br/>
          <BlockMath math="\frac{d}{dx}(x^{\color{fuchsia}{n}})={\color{fuchsia}{n}}x^{{\color{fuchsia}{n}}-1}" />
        </div>
        <div>Constant multiple rule<br/>
          <BlockMath math="\frac{d}{dx}({\color{fuchsia}{n}}{\color{orange}{f}})={\color{fuchsia}{n}}{\color{orange}{f'}}" />
        </div>
        <div>Chain rule<br/>
          <BlockMath math="\frac{d}{dx}\left({\color{orange}{f(}}{\color{lime}{g}}{\color{orange}{)}}\right)={\color{orange}{f'(}}{\color{lime}{g}}{\color{orange}{)}}{\color{lime}{g'}}" />
        </div>

        <div>Sum & difference rule<br/>
          <BlockMath math="\frac{d}{dx}({\color{orange}{f}} \pm {\color{lime}{g}})={\color{orange}{f'}} \pm {\color{lime}{g'}}" />
        </div>
        <div>Product rule<br/>
          <BlockMath math="\frac{d}{dx}({\color{orange}{f}}{\color{lime}{g}})={\color{orange}{f'}}{\color{lime}{g}}+{\color{orange}{f}}{\color{lime}{g'}}" />
        </div>
        <div>Quotient rule<br/>
          <BlockMath math="\frac{d}{dx}\left( \frac{\color{orange}{f}}{\color{lime}{g}} \right)=\frac{{\color{orange}{f'}}{\color{lime}{g}}-{\color{orange}{f}}{\color{lime}{g'}}}{{\color{lime}{g}}^2}" />
        </div>
        <div>General power rule<br/>
          <BlockMath math="\frac{d}{dx}\left({\color{orange}{f}}^{\color{fuchsia}{n}}\right)={\color{fuchsia}{n}}{\color{orange}{f}}^{{\color{fuchsia}{n}}-1}{\color{orange}{f'}}" />
        </div>

      </>;

    }
  }}
</Observe>

</div>


## Trigonometry


<div className="grid grid-flow-row md:grid-rows-3 md:grid-flow-col gap-y-3 md:gap-0">

<Observe>
  {({ setState, ...state }) => {
    switch (state.mode) {
      
      case 'regular': return <>
        <BlockMath math="\frac{d}{dx} (\sin f)=\cos(f)f'" />
        <BlockMath math="\frac{d}{dx}(\cos f)=-\sin(f)f'" />
        <BlockMath math="\frac{d}{dx}(\tan f)=\sec^2(f)f'" />

        <BlockMath math="\frac{d}{dx}(\csc f)=-\csc(f)\cot(f)f'" />
        <BlockMath math="\frac{d}{dx}(\sec f)=\sec(f)\tan(f)f'" />
        <BlockMath math="\frac{d}{dx}(\cot f)=-\csc^2(f)f'" />
      </>;

      case 'color': return <>
        <BlockMath math="\frac{d}{dx}(\sin {\color{orange}{f}})=\cos({\color{orange}{f}}){\color{orange}{f'}}" />
        <BlockMath math="\frac{d}{dx}(\cos {\color{orange}{f}})=-\sin({\color{orange}{f}}){\color{orange}{f'}}" />
        <BlockMath math="\frac{d}{dx}(\tan {\color{orange}{f}})=\sec^2({\color{orange}{f}}){\color{orange}{f'}}" />

        <BlockMath math="\frac{d}{dx}(\csc {\color{orange}{f}})=-\csc({\color{orange}{f}})\cot({\color{orange}{f}}){\color{orange}{f'}}" />
        <BlockMath math="\frac{d}{dx}(\sec {\color{orange}{f}})=\sec({\color{orange}{f}})\tan({\color{orange}{f}}){\color{orange}{f'}}" />
        <BlockMath math="\frac{d}{dx}(\cot {\color{orange}{f}})=-\csc^2({\color{orange}{f}}){\color{orange}{f'}}" />
      </>;

    }
  }}
</Observe>

</div>


## Inverse trigonometry

<div className="grid grid-flow-row md:grid-rows-3 md:grid-flow-col gap-y-3 md:gap-0">

<Observe>
  {({ setState, ...state }) => {
    switch (state.mode) {
      
      case 'regular': return <>
        <BlockMath math="\frac{d}{dx}(\sin^{-1} f)=\frac{f'}{\sqrt{1-f^2}}" />
        <BlockMath math="\frac{d}{dx}(\cos^{-1} f)=\frac{-f'}{\sqrt{1-f^2}} " />
        <BlockMath math="\frac{d}{dx}(\tan^{-1} f)=\frac{f'}{1+f^2}" />

        <BlockMath math="\frac{d}{dx}(\csc^{-1} f)=\frac{-f'}{|f|\sqrt{f^2-1}}" />
        <BlockMath math="\frac{d}{dx}(\sec^{-1} f)=\frac{f'}{|f|\sqrt{f^2-1}}" />
        <BlockMath math="\frac{d}{dx}(\cot^{-1} f)=\frac{-f'}{1+f^2}" />
      </>;

      case 'color': return <>
        <BlockMath math="\frac{d}{dx}(\sin^{-1} {\color{orange}{f}})=\frac{\color{orange}{f'}}{\sqrt{1-{\color{orange}{f}}^2}}" />
        <BlockMath math="\frac{d}{dx}(\cos^{-1} {\color{orange}{f}})=\frac{-{\color{orange}{f'}}}{\sqrt{1-{\color{orange}{f}}^2}} " />
        <BlockMath math="\frac{d}{dx}(\tan^{-1} {\color{orange}{f}})=\frac{\color{orange}{f'}}{1+{\color{orange}{f}}^2}" />

        <BlockMath math="\frac{d}{dx}(\csc^{-1} {\color{orange}{f}})=\frac{-{\color{orange}{f'}}}{|{\color{orange}{f}}|\sqrt{{\color{orange}{f}}^2-1}}" />
        <BlockMath math="\frac{d}{dx}(\sec^{-1} {\color{orange}{f}})=\frac{\color{orange}{f'}}{|{\color{orange}{f}}|\sqrt{{\color{orange}{f}}^2-1}}" />
        <BlockMath math="\frac{d}{dx}(\cot^{-1} {\color{orange}{f}})=\frac{-{\color{orange}{f'}}}{1+{\color{orange}{f}}^2}" />
      </>;

    }
  }}
</Observe>

</div>
      

## Logarithmic

<Observe>
  {({ setState, ...state }) => {
    switch (state.mode) {
      
      case 'regular': return <>
        <BlockMath math="\frac{d}{dx}(\log_nf)=\frac{\log_n(e)f'}{f}=\frac{f'}{\ln(n)f}" />
        <BlockMath math="\frac{d}{dx}(\ln f)=\frac{f'}{f}" />
      </>;

      case 'color': return <>
        <BlockMath math="\frac{d}{dx}(\log_{\color{fuchsia}{n}}{\color{orange}{f}})=\frac{\log_{\color{fuchsia}{n}}(e){\color{orange}{f'}}}{\color{orange}{f}}=\frac{\color{orange}{f'}}{\ln({\color{fuchsia}{n}}){\color{orange}{f}}}" />
        <BlockMath math="\frac{d}{dx}(\ln {\color{orange}{f}})=\frac{\color{orange}{f'}}{\color{orange}{f}}" />
      </>;

    }
  }}
</Observe>


## Inverse function *(for this, $g$ is inverse of $f$)*
<Observe>
  {({ setState, ...state }) => {

      case 'regular': return <>
        <BlockMath math="g'=\frac{1}{f'(g)}" />
      </>;

      case 'color': return <>
        <BlockMath math="{\color{lime}{g'}}=\frac{1}{{\color{orange}{f'}}({\color{lime}{g}})}" />
      </>;

    }
  }}
</Observe>

## Exponential function
<Observe>
  {({ setState, ...state }) => {
    switch (state.mode) {
      
      case 'regular': return <>
        <BlockMath math="\frac{d}{dx}(n^f)=n^f \ln(n)f'" />
        <BlockMath math="\frac{d}{dx}(e^f)=e^ff'" />
        <BlockMath math="\frac{d}{dx}(f^g)=f^g \left( \frac{gf'}{f}+g' \ln(f) \right)" />
      </>;

      case 'color': return <>
        <BlockMath math="\frac{d}{dx}({\color{fuchsia}{n}}^{\color{orange}{f}})={\color{fuchsia}{n}}^{\color{orange}{f}} \ln({\color{fuchsia}{n}}){\color{orange}{f'}}" />
        <BlockMath math="\frac{d}{dx}(e^{\color{orange}{f}})=e^{\color{orange}{f}}{\color{orange}{f'}}" className="tooltip"/>
        <BlockMath math="\frac{d}{dx}({\color{orange}{f}}^{\color{lime}{g}})={\color{orange}{f}}^{\color{lime}{g}} \left( \frac{{\color{lime}{g}}{\color{orange}{f'}}}{\color{orange}{f}}+{\color{lime}{g'}} \ln({\color{orange}{f}}) \right)" />
      </>;

    }
  }}
</Observe>
