## Indefinite integrals

<Observe>
  {({ setState, ...state }) => {
    switch (state.mode) {
      case 'regular':
        return <BlockMath math="\begin{align*}
        &\int 0 \space dx=C \\
        &\int dx=x+C \\
        &\int n f \space dx=n\int f \space dx \\
        &\int (f\pm g) \space dx=\int f \space dx\pm\int g \space dx\\
        &\int x^n\space dx=\begin{cases}
          \frac{x^{n+1}}{n+1}+C &n\neq-1 \\
          \ln|x|+C&n=-1
        \end{cases} \\
        &\int f'(g)g'\space dx=f(g) 
        \end{align*}" 
    />;
    
      case 'color':
        return  <BlockMath math="\begin{align*}
        &\int 0 \space dx=C \\
        &\int dx=x+C \\
        &\int {\color{fuchsia}{n}} {\color{orange}{f}} \space dx={\color{fuchsia}{n}}\int {\color{orange}{f}} \space dx \\
        &\int ({\color{orange}{f}}\pm {\color{orange}{g}}) \space dx=\int {\color{orange}{f}} \space dx\pm\int {\color{orange}{g}} \space dx\\
        &\int {\color{aqua}{x}}^{\color{fuchsia}{n}}\space dx=\begin{cases}
        \frac{{\color{aqua}{x}}^{{\color{fuchsia}{n}}+1}}{{\color{fuchsia}{n}}+1}+C &{\color{fuchsia}{n}}\neq-1 \\
        \ln|{\color{aqua}{x}}|+C&{\color{fuchsia}{n}}=-1
        \end{cases} \\
        &\int {\color{orange}{f'(}}g{\color{orange}{)}}g'\space dx={\color{orange}{f(}}g{\color{orange}{)}} 
        \end{align*}" 
    />
    }

  }


  }
</Observe>

### Inverse trigonometry
<Observe>
  {({ setState, ...state }) => {
    switch (state.mode) {
      case 'regular':
        return <BlockMath math="\begin{align*}
        &\int \frac{du}{\sqrt{n^{2}-u^{2}}}=\sin^{-1}\frac{u}{n}+C&-\\
        &-&\int& \frac{du}{u\sqrt{u^{2}-n^{2}}}=\frac{1}{n}\sec^{-1}\frac{u}{n}+C\\
        &\int \frac{du}{n^{2}+u^{2}}=\frac{1}{n}\tan^{-1}\frac{u}{n}+C&-
        \end{align*}" 
    />;
    
      case 'color':
        return  <BlockMath math="\begin{align*}
        &\int \frac{du}{\sqrt{{\color{fuchsia}{n}}^{2}-{\color{aqua}{u}}^{2}}}=\sin^{-1}\frac{\color{aqua}{u}}{\color{fuchsia}{n}}+C&-\\
        &-&\int& \frac{du}{{\color{aqua}{u}}\sqrt{{\color{aqua}{u}}^{2}-{\color{fuchsia}{n}}^{2}}}=\frac{1}{\color{fuchsia}{n}}\sec^{-1}\frac{\color{aqua}{u}}{\color{fuchsia}{n}}+C\\
        &\int \frac{du}{{\color{fuchsia}{n}}^{2}+{\color{aqua}{u}}^{2}}=\frac{1}{\color{fuchsia}{n}}\tan^{-1}\frac{\color{aqua}{u}}{\color{fuchsia}{n}}+C&-
        \end{align*}" 
    />
      }
    }
  }
</Observe>

### Exponential
<Observe>
  {({ setState, ...state }) => {
    switch (state.mode) {
      case 'regular':
        return <BlockMath math="\begin{align*}
        &\int n^u\space du=\frac{n^u}{\ln n}+C\\
        &\int e^u\space du=e^u+C
        \end{align*}" 
    />;
    
      case 'color':
        return  <BlockMath math="\begin{align*}
        &\int {\color{fuchsia}{n}}^{\color{aqua}{u}}\space du=\frac{{\color{fuchsia}{n}}^{\color{aqua}{u}}}{\ln {\color{fuchsia}{n}}}+C\\
        &\int e^{\color{aqua}{u}}\space du=e^{\color{aqua}{u}}+C
        \end{align*}" 
    />
      }
    }
  }
</Observe>

## Definite integrals
$a$ and $b$ are the lower and upper bounds, respectively.

<Observe>
  {({ setState, ...state }) => {
    switch (state.mode) {
      case 'regular':
        return <BlockMath math="\begin{align*}
        &\int_a^bf\ dx=F\bigg|_a^b=F(b)-F(a)\\
        &\int_a^bf\ dx=-\int_b^af\ dx\\
        &\int_a^af\ dx=0\\
        &\int_a^b nf\ dx=n\int_a^b f\ dx\\
        &\int_a^b(f\pm g)\ dx=\int_a^bf\ dx \pm \int_a^b g\ dx\\
        &\int_a^b f\ dx=\int_a^d f\ dx+\int_d^b f\ dx\\
        \end{align*}" 
    />;
    
      case 'color':
        return <BlockMath math="\begin{align*}
        &\int_a^b{\color{orange}{f}}\ dx={\color{orange}{F}}\bigg|_a^b={\color{orange}{F(}}b{\color{orange}{)}}-{\color{orange}{F(}}a{\color{orange}{)}}\\
        &\int_a^b{\color{orange}{f}}\ dx=-\int_b^a{\color{orange}{f}}\ dx\\
        &\int_a^a{\color{orange}{f}}\ dx=0\\
        &\int_a^b {\color{fuchsia}{n}}{\color{orange}{f}}\ dx={\color{fuchsia}{n}}\int_a^b {\color{orange}{f}}\ dx\\
        &\int_a^b({\color{orange}{f}}\pm {\color{orange}{g}})\ dx=\int_a^b{\color{orange}{f}}\ dx \pm \int_a^b {\color{orange}{g}}\ dx\\
        &\int_a^b {\color{orange}{f}}\ dx=\int_a^d {\color{orange}{f}}\ dx+\int_d^b {\color{orange}{f}}\ dx\\
        \end{align*}" 
    />
      }
    }
  }
</Observe>
